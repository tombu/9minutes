78:       request.on_complete do |response|

79:         if response.success?

80:           puts "=> lastfm: '#{current_class_name}.#{method}' (#{response.time}s)"

81:           

82:           hash = Hashie::Mash.new(JSON.parse(response.body))

86:           block.call( hash ) unless block.nil?
